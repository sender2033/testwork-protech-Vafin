Настройка milrotik wan
```
[admin@MikroTik] > /interface ethernet set [find name=ether1] name=wan
[admin@MikroTik] > /interface ethernet set [find name=ether2] name=trunk-to-r2
[admin@MikroTik] > /interface vlan
[admin@MikroTik] /interface vlan> add name=VLAN10 interface=trunk-to-r2 vlan-id=10
[admin@MikroTik] /interface vlan> add name=VLAN20 interface=trunk-to-r2 vlan-id=20
[admin@MikroTik] /interface vlan> /ip address
[admin@MikroTik] /ip address> add address=172.16.1.1/24 interface=VLAN10 comment="Client GW"
[admin@MikroTik] /ip address> add address=172.16.2.1/24 interface=VLAN20 comment="Server GW"
[admin@MikroTik] /ip address>
[admin@MikroTik] /ip address> /ip firewall nat
[admin@MikroTik] /ip firewall nat> add chain=srcnat out-interface=wan action=masquerade
[admin@MikroTik] /ip firewall nat> /ip route
[admin@MikroTik] /ip route> add dst-address=0.0.0.0/0 gateway=192.168.209.129
```

```
[admin@MikroTik] > /interface bridge
[admin@MikroTik] /interface bridge> add name=bridge-lan vlan-filtering=yes
[admin@MikroTik] /interface bridge> /interface bridge port
[admin@MikroTik] /interface bridge port> add bridge=bridge-lan interface=ether2 ingress-filtering=yes frame-types=admit-all
[admin@MikroTik] /interface bridge port> add bridge=bridge-lan interface=ether3 pvid=10
[admin@MikroTik] /interface bridge port> add bridge=bridge-lan interface=ether4 pvid=20
[admin@MikroTik] /interface bridge port> add bridge=bridge-lan interface=ether1 pvid=20
[admin@MikroTik] /interface bridge port> add bridge=bridge-lan interface=ether5 pvid=20
[admin@MikroTik] /interface bridge port> /interface bridge vlan
[admin@MikroTik] /interface bridge vlan> add bridge=bridge-lan vlan-ids=10 tagged=bridge-lan,ether2 untagged=ether3
[admin@MikroTik] /interface bridge vlan>
[admin@MikroTik] /interface bridge vlan> add bridge=bridge-lan vlan-ids=20 tagged=bridge-lan,ether2 untagged=ether4,ether1,ether5
[admin@MikroTik] /interface bridge vlan> /interface vlan
[admin@MikroTik] /interface vlan> add name=VLAN10 interface=bridge-lan vlan-id=10
[admin@MikroTik] /interface vlan> add name=VLAN20 interface=bridge-lan vlan-id=20
[admin@MikroTik] /interface vlan> /ip address
[admin@MikroTik] /ip address> add address=172.16.1.1/24 interface=VLAN10
[admin@MikroTik] /ip address> add address=172.16.2.1/24 interface=VLAN20
[admin@MikroTik] /ip address> /ip dns
[admin@MikroTik] /ip dns> set allow-remote-requests=yes servers=8.8.8.8,1.1.1.1
[admin@MikroTik] /ip dns> /interface bridge set bridge-lan vlan-filtering=yes
[admin@MikroTik] /ip dns>

```
